```python
def solution(n, lost, reserve):
    copied = []
    for man in reserve:
        if man in lost:
            copied.append(man)
            lost.remove(man)
    for copy in copied:
        reserve.remove(copy)
    lost.sort()
    for man in reserve:
        if man - 1 in lost:
            lost.remove(man-1)
        elif man + 1 in lost:
            lost.remove(man+1)
            
    return n - len(lost)
```
