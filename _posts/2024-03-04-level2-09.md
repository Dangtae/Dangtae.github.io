def solution(n, words):
    word_list = [words[0]]
    for word in words[1:]:
        word_list.append(word)
        if word in word_list[:-1] or word_list[-2][-1] != word[0]:
            if len(word_list) % n == 0:
                number = n
                turn = len(word_list) // n
            else:
                number = len(word_list) % n
                turn = len(word_list) // n + 1
        
            return [number, turn]
        if len(words) == len(word_list):
            return [0,0]
