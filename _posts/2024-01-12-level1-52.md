```python
import numpy as np

def solution(board, moves):
    a = np.transpose(board)
    a = a.tolist()
    dolls = []
    count = 0
    for move in moves:
        for doll in a[move-1]:
            if doll != 0:
                a[move-1].remove(doll)
                if dolls:
                    if dolls[-1] != doll:
                        dolls.append(doll)
                        break
                    else:
                        dolls.pop()
                        count += 2
                        break
                else:
                    dolls.append(doll)
                    break
    return count
```
