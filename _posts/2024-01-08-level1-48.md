```python
def solution(lottos, win_nums):
    prize = {6: 1, 5: 2, 4: 3, 3: 4, 2: 5, 1: 6, 0: 6}
    count = 0 
    result = []
    for lotto in lottos:
        if lotto in win_nums:
            count += 1
    a = lottos.count(0)
    best = count + a
    worst = count
    result.append(prize[best])
    result.append(prize[worst])
    
    return result
```
