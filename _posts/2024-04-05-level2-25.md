def solution(s):
    num_count = {}
    s = s.replace('{','').replace('}','').split(',')
    s_set = set(s)
    for num in s_set:
        num_count[int(num)] = s.count(num)
    result = sorted(num_count, key = lambda x: num_count[x], reverse = True)
    
    return result
